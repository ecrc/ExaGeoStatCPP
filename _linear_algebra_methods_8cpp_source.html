298 template<typename T>
299 void LinearAlgebraMethods<T>::GenerateObservationsVector(Configurations &aConfigurations, ExaGeoStatData<T> &aData,
300  Locations<T> *apLocation1, Locations<T> *apLocation2,
301  Locations<T> *apLocation3, const int &aDistanceMetric,
302  const kernels::Kernel<T> &aKernel) {
303 
304  // Check for initialize the Chameleon context.
305  if (!this->mpContext) {
306  throw std::runtime_error(
307  "ExaGeoStat hardware is not initialized, please use 'ExaGeoStatHardware(computation, cores_number, gpu_numbers);'.");
308  }
309 
310  const int n = aConfigurations.GetProblemSize();
311  int seed = aConfigurations.GetSeed();
312  int initial_seed[4] = {seed, seed, seed, 1};
313  T time_facto = 0.0, time_trmm = 0.0, matrix_gen_time = 0.0, flops = 0;
314 
315  // Create a Chameleon sequence, if not initialized before through the same descriptors
316  RUNTIME_request_t request_array[2] = {RUNTIME_REQUEST_INITIALIZER, RUNTIME_REQUEST_INITIALIZER};
317  RUNTIME_sequence_t *sequence;
318  if (!aData.GetDescriptorData()->GetSequence()) {
319  ExaGeoStatCreateSequence(&sequence);
320  aData.GetDescriptorData()->SetSequence(sequence);
321  aData.GetDescriptorData()->SetRequest(request_array);
322  } else {
323  sequence = (RUNTIME_sequence_t *) aData.GetDescriptorData()->GetSequence();
324  }
325  auto *CHAM_descC = aData.GetDescriptorData()->GetDescriptor(CHAMELEON_DESCRIPTOR, DESCRIPTOR_C).chameleon_desc;
326  //normal random generation of e -- ei~N(0, 1) to generate Z
327  auto *randomN = new T[n];
328  LAPACKE_dlarnv(3, initial_seed, n, (double *) randomN);
329 
330  //Generate the co-variance matrix C
331  auto *theta = new T[aConfigurations.GetInitialTheta().size()];
332  for (int i = 0; i < aConfigurations.GetInitialTheta().size(); i++) {
333  theta[i] = aConfigurations.GetInitialTheta()[i];
334  }
335 
336  VERBOSE("Initializing Covariance Matrix (Synthetic Dataset Generation Phase).....")
337  int upper_lower = EXAGEOSTAT_LOWER;
338  START_TIMING(matrix_gen_time);
339 
340  this->CovarianceMatrixCodelet(*aData.GetDescriptorData(), CHAM_descC, upper_lower, apLocation1, apLocation2,
341  apLocation3, theta, aDistanceMetric, &aKernel);
342  ExaGeoStatSequenceWait(sequence);
343  STOP_TIMING(matrix_gen_time);
344  VERBOSE("Done.")
345 
346  //Copy randomN to Z
347  VERBOSE("Generate Normal Random Distribution Vector Z (Synthetic Dataset Generation Phase) .....")
348  auto *CHAM_descZ = aData.GetDescriptorData()->GetDescriptor(CHAMELEON_DESCRIPTOR, DESCRIPTOR_Z).chameleon_desc;
349  CopyDescriptorZ(*aData.GetDescriptorData(), CHAM_descZ, randomN);
350  VERBOSE("Done.")
351 
352  //Cholesky factorization for the Co-variance matrix C
353  VERBOSE("Cholesky factorization of Sigma (Synthetic Dataset Generation Phase) .....")
354  START_TIMING(time_facto);
355  ExaGeoStatPotrfTile(EXAGEOSTAT_LOWER, CHAM_descC, aConfigurations.GetBand(), nullptr, nullptr, 0, 0);
356  STOP_TIMING(time_facto);
357  flops = flops + flops_dpotrf(n);
358  VERBOSE("Done.")
359 
360  //Triangular matrix-matrix multiplication
361  VERBOSE("Triangular matrix-matrix multiplication Z=L.e (Synthetic Dataset Generation Phase) .....")
362  START_TIMING(time_trmm);
363  ExaGeoStatTrmmTile(EXAGEOSTAT_LEFT, EXAGEOSTAT_LOWER, EXAGEOSTAT_NO_TRANS, EXAGEOSTAT_NON_UNIT, 1, CHAM_descC,
364  CHAM_descZ);
365  STOP_TIMING(time_trmm);
366  flops = flops + flops_dtrmm(ChamLeft, n, CHAM_descZ->n);
367  VERBOSE("Done.")
368 
369  if (aConfigurations.GetKernelName() == "UnivariateMaternNonGaussian") {
370  //Gaussian to non-gaussian transformation
371  VERBOSE("Convert Z Gaussian to non-Gaussian (Synthetic Dataset Generation Phase) .....")
372  ExaGeoStatGaussianToNonTileAsync(*aData.GetDescriptorData(), CHAM_descZ, theta);
373  VERBOSE("Done.")
374  }
375  delete[] theta;
376  const int P = aConfigurations.GetP();
377  if (aConfigurations.GetLogger()) {
378  T *pMatrix;
379  VERBOSE("Writing generated data to the disk (Synthetic Dataset Generation Phase) .....")
380 #ifdef CHAMELEON_USE_MPI
381  pMatrix = new T[n];
382  string path = aConfigurations.GetLoggerPath();
383  ExaGeoStatDesc2Lap(pMatrix, n, CHAM_descZ, EXAGEOSTAT_UPPER_LOWER);
384  if ( CHAMELEON_Comm_rank() == 0 ){
385  helpers::DiskWriter<T>::WriteVectorsToDisk(*pMatrix, n, P, path, *apLocation1);
386  }
387  delete[] pMatrix;
388 #else
389  pMatrix = (T *) CHAM_descZ->mat;
390  string path = aConfigurations.GetLoggerPath();
391  helpers::DiskWriter<T>::WriteVectorsToDisk(*pMatrix, n, P, path, *apLocation1);
392 #endif
393  VERBOSE("Done.")
394  }
395 
396  ExaGeoStatLaSetTile(EXAGEOSTAT_UPPER_LOWER, 0, 0, CHAM_descC);
397  delete[] randomN;
398  VERBOSE("Done Z Vector Generation Phase. (Chameleon Synchronous)")
399 
400  int total_flops = flops / 1e9 / (time_facto + time_trmm);
401  LOGGER(" ---- Facto Time: " << time_facto)
402  LOGGER(" ---- dtrmm Time: " << time_trmm)
403  LOGGER(" ---- Matrix Generation Time: " << matrix_gen_time)
404  LOGGER(" ---- Total Time: " << time_facto + time_trmm)
405  LOGGER(" ---- Gflop/s: " << total_flops)
