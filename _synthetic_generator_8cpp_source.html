96 template<typename T>
97 void SyntheticGenerator<T>::GenerateLocations(const int &aN, const int &aTimeSlot, const Dimension &aDimension,
98  Locations<T> &aLocations) {
99 
100  aLocations.SetSize(aN);
101  int index = 0;
102  aLocations.SetDimension(aDimension);
103 
104  int rootN;
105  if (aDimension == Dimension3D) {
106  //Cubic root.
107  rootN = ceil(cbrt(aN));
108  } else {
109  //Square root.
110  rootN = ceil(sqrt(aN));
111  }
112 
113  int *grid = new int[rootN]();
114  for (auto i = 0; i < rootN; i++) {
115  grid[i] = i + 1;
116  }
117 
118  double range_low = -0.4, range_high = 0.4;
119 
120  for (auto i = 0; i < rootN && index < aN; i++) {
121  for (auto j = 0; j < rootN && index < aN; j++) {
122  if (aDimension == Dimension3D) {
123  for (auto k = 0; k < rootN && index < aN; k++) {
124  aLocations.GetLocationX()[index] =
125  (grid[i] - 0.5 + UniformDistribution(range_low, range_high)) / rootN;
126  aLocations.GetLocationY()[index] =
127  (grid[j] - 0.5 + UniformDistribution(range_low, range_high)) / rootN;
128  aLocations.GetLocationZ()[index] =
129  (grid[k] - 0.5 + UniformDistribution(range_low, range_high)) / rootN;
130  index++;
131  }
132  } else {
133  aLocations.GetLocationX()[index] = (grid[i] - 0.5 + UniformDistribution(range_low, range_high)) / rootN;
134  aLocations.GetLocationY()[index] = (grid[j] - 0.5 + UniformDistribution(range_low, range_high)) / rootN;
135  if (aDimension == DimensionST) {
136  aLocations.GetLocationZ()[index] = 1.0;
137  }
138  index++;
139  }
140  }
141  }
142  delete[] grid;
143  if (aDimension != DimensionST) {
144  SortLocations(aN, aDimension, aLocations);
145  } else {
146  for (auto j = 1; j < aTimeSlot; j++) {
147  for (auto i = 0; i < aN; i++) {
148  aLocations.GetLocationX()[i + j * aN] = aLocations.GetLocationX()[i];
149  aLocations.GetLocationY()[i + j * aN] = aLocations.GetLocationY()[i];
150  aLocations.GetLocationZ()[i + j * aN] = (double) (j + 1);
151  }
152  }
153  }
154 }

156 template<typename T>
157 double SyntheticGenerator<T>::UniformDistribution(const double &aRangeLow, const double &aRangeHigh) {
158  double myRand = (double) rand() / (double) (1.0 + RAND_MAX);
159  double range = aRangeHigh - aRangeLow;
160  return (myRand * range) + aRangeLow;
161 }

163 template<typename T>
164 void SyntheticGenerator<T>::SortLocations(const int &aN, const Dimension &aDimension, Locations<T> &aLocations) {
165 
166  // Some sorting, required by spatial statistics code
167  uint16_t x, y, z;
168  uint64_t vectorZ[aN];
169 
170  // Encode data into vector z
171  for (auto i = 0; i < aN; i++) {
172  x = (uint16_t) (aLocations.GetLocationX()[i] * (double) UINT16_MAX + .5);
173  y = (uint16_t) (aLocations.GetLocationY()[i] * (double) UINT16_MAX + .5);
174  if (aDimension != Dimension2D) {
175  z = (uint16_t) (aLocations.GetLocationZ()[i] * (double) UINT16_MAX + .5);
176  } else {
177  z = (uint16_t) 0.0;
178  }
179  vectorZ[i] = (SpreadBits(z) << 2) + (SpreadBits(y) << 1) + SpreadBits(x);
180  }
181  // Sort vector z
182  std::sort(vectorZ, vectorZ + aN, CompareUint64);
183 
184  // Decode data from vector z
185  for (auto i = 0; i < aN; i++) {
186  x = ReverseSpreadBits(vectorZ[i] >> 0);
187  y = ReverseSpreadBits(vectorZ[i] >> 1);
188  z = ReverseSpreadBits(vectorZ[i] >> 2);
189  aLocations.GetLocationX()[i] = (double) x / (double) UINT16_MAX;
190  aLocations.GetLocationY()[i] = (double) y / (double) UINT16_MAX;
191  if (aDimension == Dimension3D) {
192  aLocations.GetLocationZ()[i] = (double) z / (double) UINT16_MAX;
193  }
194  }
195 }

197 template<typename T>
198 uint64_t SyntheticGenerator<T>::SpreadBits(uint64_t aInputByte) {
199  aInputByte &= 0x000000000000ffff;
200  // aInputByte = ---- ---- ---- ---- ---- ---- ---- ---- ---- ---- ---- ---- fedc ba98 7654 3210
201  aInputByte = (aInputByte ^ (aInputByte << 24)) & 0x000000ff000000ff;
202  // aInputByte = ---- ---- ---- ---- ---- ---- fedc ba98 ---- ---- ---- ---- ---- ---- 7654 3210
203  aInputByte = (aInputByte ^ (aInputByte << 12)) & 0x000f000f000f000f; //000 7000f000f000f
204  // aInputByte = ---- ---- ---- fedc ---- ---- ---- ba98 ---- ---- ---- 7654 ---- ---- ---- 3210
205  aInputByte = (aInputByte ^ (aInputByte << 6)) & 0x0303030303030303; //0 0001 0 0011 0 0011 0 0011 0
206  // aInputByte = ---- --fe ---- --dc ---- --ba ---- --98 ---- --76 ---- --54 ---- --32 ---- --10
207  aInputByte = (aInputByte ^ (aInputByte << 3)) & 0x1111111111111111;
208  // aInputByte = ---f ---e ---d ---c ---b ---a ---9 ---8 ---7 ---6 ---5 ---4 ---3 ---2 ---1 ---0
209  return aInputByte;
210 }

212 template<typename T>
213 uint64_t SyntheticGenerator<T>::ReverseSpreadBits(uint64_t aInputByte) {
214 
215  aInputByte &= 0x1111111111111111;
216  // aInputByte = ---f ---e ---d ---c ---b ---a ---9 ---8 ---7 ---6 ---5 ---4 ---3 ---2 ---1 ---0
217  aInputByte = (aInputByte ^ (aInputByte >> 3)) & 0x0303030303030303;
218  // aInputByte = ---- --fe ---- --dc ---- --ba ---- --98 ---- --76 ---- --54 ---- --32 ---- --10
219  aInputByte = (aInputByte ^ (aInputByte >> 6)) & 0x000f000f000f000f;
220  // aInputByte = ---- ---- ---- fedc ---- ---- ---- ba98 ---- ---- ---- 7654 ---- ---- ---- 3210
221  aInputByte = (aInputByte ^ (aInputByte >> 12)) & 0x000000ff000000ff;
222  // aInputByte = ---- ---- ---- ---- ---- ---- fedc ba98 ---- ---- ---- ---- ---- ---- 7654 3210
223  aInputByte = (aInputByte ^ (aInputByte >> 24)) & 0x000000000000ffff;
224  // aInputByte = ---- ---- ---- ---- ---- ---- ---- ---- ---- ---- ---- ---- fedc ba98 7654 3210
225  return aInputByte;
