22 template<typename T>
23 DescriptorData<T>::~DescriptorData() {
24 
25  ExaGeoStatDescriptor<T> exaGeoStatDescriptor;
26  // Destroy descriptors.
27  const std::string &chameleon = "_CHAMELEON";
28  for (const auto &pair: this->mDictionary) {
29  const std::string &key = pair.first;
30  if (key.find("CHAMELEON") != std::string::npos && pair.second != nullptr) {
31  exaGeoStatDescriptor.DestroyDescriptor(common::CHAMELEON_DESCRIPTOR, pair.second);
32 #ifdef EXAGEOSTAT_USE_HICMA
33  // Since there are converted descriptors from Chameleon to Hicma, which have the same memory address.
34  // So, by deleting the owner which is Chameleon, no need to delete hicma. Therefore, we remove the row of that descriptor.
35  std::string converted_chameleon = key.substr(0, key.length() - chameleon.length());
36  const std::string &desc = converted_chameleon + "_CHAM_HIC";
37  if (this->mDictionary.find(desc) != this->mDictionary.end()) {
38  delete (HICMA_desc_t *)
39  this->mDictionary[converted_chameleon + "_CHAM_HIC"];
40  this->mDictionary.erase(converted_chameleon + "_CHAM_HIC");
41  }
42 #endif
43  } else if (key.find("HICMA") != std::string::npos && pair.second != nullptr) {
44  exaGeoStatDescriptor.DestroyDescriptor(common::HICMA_DESCRIPTOR, pair.second);
45  }
46  }
47  this->mDictionary.clear();
48  if (this->mpSequence) {
49  CHAMELEON_Sequence_Destroy((RUNTIME_sequence_t *) this->mpSequence);
50  }
51 }

75 template<typename T>
76 HICMA_desc_t *DescriptorData<T>::ConvertChameleonToHicma(CHAM_desc_t *apChameleonDesc) {
77 
78  // Create a new HICMA descriptor
79  auto *hicma_desc = new HICMA_desc_t;
80 
81  // Set function pointers in HICMA descriptor
82  hicma_desc->get_blkaddr = hicma_getaddr_ccrb;
83  hicma_desc->get_blkldd = hicma_getblkldd_ccrb;
84  hicma_desc->get_rankof = hicma_getrankof_2d;
85 
86  // Set sizes and offsets for memory copy
87  size_t hicma_desc_total_size = 184;
88  size_t chameleon_desc_total_size = 200;
89  // Size of the common members between Hicma_desc and Chameleon_desc
90  size_t common_total_size = 3 * sizeof(size_t) + 30 * sizeof(int);
91  // Skip the size of function pointers.
92  size_t hicma_offset = hicma_desc_total_size - (common_total_size + sizeof(void *));
93  size_t chameleon_offset = chameleon_desc_total_size - (common_total_size + sizeof(void *));
94 
95  // Copy common data from CHAMELEON descriptor to HICMA descriptor
96  memcpy(((char *) hicma_desc) + hicma_offset, ((char *) apChameleonDesc) + chameleon_offset, common_total_size);
97  // Set additional data in HICMA descriptor
98  hicma_desc->mat = apChameleonDesc->mat;
99  hicma_desc->schedopt = apChameleonDesc->schedopt;
100 
101  return hicma_desc;
102 }

107 BaseDescriptor
108 DescriptorData<T>::GetDescriptor(const DescriptorType &aDescriptorType, const DescriptorName &aDescriptorName) {
109 
110  BaseDescriptor descriptor{};
111  if (aDescriptorType == CHAMELEON_DESCRIPTOR) {
112  if (this->mDictionary.find(GetDescriptorName(aDescriptorName) + "_CHAMELEON") == this->mDictionary.end()) {
113  descriptor.chameleon_desc = nullptr;
114  }
115  descriptor.chameleon_desc = (CHAM_desc_t *) this->mDictionary[GetDescriptorName(aDescriptorName) +
116  "_CHAMELEON"];
117  } else {
118 #ifdef EXAGEOSTAT_USE_HICMA
119  if (this->mDictionary.find(GetDescriptorName(aDescriptorName) + "_HICMA") != this->mDictionary.end()) {
120  descriptor.hicma_desc = (HICMA_desc_t *)
121  this->mDictionary[GetDescriptorName(aDescriptorName) + "_HICMA"];
122  } else if (this->mDictionary.find(GetDescriptorName(aDescriptorName) + "_CHAM_HIC") !=
123  this->mDictionary.end()) {
124  descriptor.hicma_desc = (HICMA_desc_t *)
125  this->mDictionary[GetDescriptorName(aDescriptorName) + "_CHAM_HIC"];
126  } else if (this->mDictionary.find(GetDescriptorName(aDescriptorName) + "_CHAMELEON") !=
127  this->mDictionary.end()) {
128  descriptor.hicma_desc = this->ConvertChameleonToHicma(
129  (CHAM_desc_t *) this->mDictionary[GetDescriptorName(aDescriptorName) + "_CHAMELEON"]);
130  this->mDictionary[GetDescriptorName(aDescriptorName) + "_CHAM_HIC"] = descriptor.hicma_desc;
131  } else {
132  descriptor.hicma_desc = nullptr;
133  }
134 #else
135  throw std::runtime_error("To use HiCMA descriptor you need to enable EXAGEOSTAT_USE_HICMA!");
136 #endif
137  }
138  return descriptor;
139 }
