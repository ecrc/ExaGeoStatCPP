258  const static std::set<std::string> availableKernels = []() {
259  // This set stores the kernel names.
260  std::set<std::string> kernelNames;
261  // This string stores the directory path where the kernel files are located.
262  // The path is obtained by using the __FILE__ macro to get the full path of the current source file,
263  // and then navigating two levels up to reach the directory that contains the kernel files.
264  const std::string directoryPath = KERNELS_PATH;
265  // This loop iterates through all the files in the directory and extracts the kernel names.
266  for (const auto &entry: std::filesystem::directory_iterator(directoryPath)) {
267  // This checks if the current entry is a regular file.
268  if (entry.is_regular_file()) {
269  // This string stores the filename of the current entry.
270  const std::string filename = entry.path().filename().string();
271  // This string stores the file extension of the current entry.
272  const std::string extension = std::filesystem::path(filename).extension().string();
273  // This string stores the kernel name extracted from the filename.
274  const std::string kernelName = filename.substr(0, filename.size() - extension.size());
275  // This adds the kernel name to the kernelNames set.
276  kernelNames.insert(kernelName);
277  // This blo/ck of code converts the kernel name to lowercase and adds underscores before each capital letter.
278  std::string lowercaseName;
279  for (std::size_t i = 0; i < kernelName.size(); ++i) {
280  if (std::isupper(kernelName[i])) {
281  // Avoid adding _ in the beginning of the name.
282  if (i != 0) {
283  lowercaseName += '_';
284  }
285  lowercaseName += static_cast<char>(std::tolower(kernelName[i]));
286  } else {
287  lowercaseName += kernelName[i];
288  }
289  }
290  // This adds the lowercase kernel name to the kernelNames set.
291  kernelNames.insert(lowercaseName);
292  }
293  }
294  return kernelNames;
295  }();
