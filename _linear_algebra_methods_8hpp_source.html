596  static void core_gaussian_to_non(T *Z, T *apLocalTheta, int aSize) {
597 
598  double xi = apLocalTheta[2];
599  double omega = apLocalTheta[3];
600  double g = apLocalTheta[4];
601  double h = apLocalTheta[5];
602 
603  int i;
604  if (h < 0) {
605  LOGGER("The kurtosis parameter cannot be negative")
606  return;
607  }
608  if (g == 0) {
609  for (i = 0; i < aSize; i++)
610  Z[i] = xi + omega * Z[i] * (exp(0.5 * h * pow(Z[i], 2)));
611  } else {
612  for (i = 0; i < aSize; i++)
613  Z[i] = xi + omega * (exp(g * Z[i]) - 1) * (exp(0.5 * h * pow(Z[i], 2))) / g;
614  }
615  }
