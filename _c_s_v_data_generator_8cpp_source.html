86 CSVDataGenerator<T>::ReadData(Configurations &aConfigurations,
87  vector<T> &aMeasurementsMatrix, vector<T> &aXLocations,
88  vector<T> &aYLocations, vector<T> &aZLocations) {
89 
90  //Check if the user entered a valid path for the CSV file.
91  if (aConfigurations.GetDataPath().empty()) {
92  throw runtime_error("Please enter the path to data file.");
93  }
94 
95  string data_path = aConfigurations.GetDataPath();
96  Dimension dimension = aConfigurations.GetDimension();
97  int p = aConfigurations.GetP();
98  ifstream file;
99 
100  file.open(data_path, ios::in);
101 
102  //Throw error if unable to open file.
103  if (!file.is_open()) {
104  throw runtime_error("Cannot read locations file: " + data_path);
105  }
106 
107  //Keep track of number of lines.
108  int index = 0;
109  string line;
110  //Loop through the lines
111  while (getline(file, line)) {
112  istringstream iss(line);
113  string token;
114 
115  // Split each line into tokens using ',' as the delimiter
116  if (getline(iss, token, ',')) {
117  aXLocations.push_back(stod(token));
118  }
119  if (getline(iss, token, ',')) {
120  aYLocations.push_back(stod(token));
121  }
122  if (getline(iss, token, ',')) {
123  //If it's a 2D locations data, the last values of the lines should be the measurement values.
124  //If it's 3D locations data, the third value of the line should be the Z coordinate.
125  //If it's ST location data, the third value of the line should be the Time coordinate
126  if (dimension == Dimension2D) {
127  aMeasurementsMatrix.push_back(stod(token));
128  //if p == 2, the third and fourth values of each line are saved in the Measurements Matrix.
129  if (p == 2) {
130  if (getline(iss, token, ',')) {
131  aMeasurementsMatrix.push_back(stod(token));
132  } else {
133  throw runtime_error(
134  "The data P in the provided file isn't consistent with the Kernel's P.");
135  }
136  }
137  if (p == 3) {
138  //if p == 3, the third, fourth, and fifth values of each line are saved in the Measurements Matrix.
139  if (getline(iss, token, ',')) {
140  aMeasurementsMatrix.push_back(stod(token));
141  }
142  if (getline(iss, token, ',')) {
143  aMeasurementsMatrix.push_back(stod(token));
144  } else {
145  throw runtime_error(
146  "The data P in the provided file isn't consistent with the Kernel's P.");
147  }
148  }
149  } else {
150  aZLocations.push_back(stod(token));
151  }
152  }
153 
154  if (getline(iss, token, ',')) {
155  if (dimension != Dimension2D) {
156  aMeasurementsMatrix.push_back(stod(token));
157  //if p == 2, the fourth and fifth values of each line are saved in the Measurements Matrix.
158  if (p == 2) {
159  if (getline(iss, token, ',')) {
160  aMeasurementsMatrix.push_back(stod(token));
161  } else {
162  throw runtime_error(
163  "The data P in the provided file isn't consistent with the Kernel's P.");
164  }
165  }
166  if (p == 3) {
167  //if p == 3, the fourth, fifth, and sixth values of each line are saved in the Measurements Matrix.
168  if (getline(iss, token, ',')) {
169  aMeasurementsMatrix.push_back(stod(token));
170  }
171  if (getline(iss, token, ',')) {
172  aMeasurementsMatrix.push_back(stod(token));
173  } else {
174  throw runtime_error(
175  "The data P in the provided file isn't consistent with the Kernel's P.");
176  }
177  }
178  } else {
179  throw runtime_error(
180  "The data dimensions in the provided file isn't consistent with the dimensions input.");
181  }
182  } else if (dimension != Dimension2D) {
183  throw runtime_error("The data dimensions in the provided file isn't consistent with the dimensions input.");
184  }
185  index++;
186  }
187 
188  //problem size is equal to the number of the CSV lines * P / aConfigurations.GetTimeSlot().
189  aConfigurations.SetProblemSize(index * p / aConfigurations.GetTimeSlot());
190 
191  file.close();
