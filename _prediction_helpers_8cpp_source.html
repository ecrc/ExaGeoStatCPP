222 int PredictionHelpers<T>::SortInplace(int aN, Locations<T> &aLocations, T *apZ) {
223 
224  int i;
225  int j;//new_j, tmp_j;
226  int count = aN;
227  int dimension_number = 2;
228  T *point = new T[dimension_number * count];
229  T *ptr1;
230  T *minmax; // min is stored in lower part, max is stored in upper part
231 
232  for (i = 0; i < count; i++) {
233  point[i] = aLocations.GetLocationX()[i];
234  point[i + count] = aLocations.GetLocationY()[i];
235  }
236 
237  minmax = new T[2 * count];
238 
239  for (i = 0; i < dimension_number; i++) {
240  ptr1 = point + i * count; // i-th dimension
241  minmax[i] = ptr1[0];
242  minmax[i + dimension_number] = minmax[i];
243  for (j = 1; j < count; j++) {
244  if (minmax[i] > ptr1[j])
245  minmax[i] = ptr1[j];
246  else if (minmax[i + dimension_number] < ptr1[j])
247  minmax[i + dimension_number] = ptr1[j];
248  }
249  }
250 
251  // Now minmax[0:dimension_number] and minmax[dimension_number:2*dimension_number] store minimal and maximal
252  // values of coordinates
253  auto uint_point = new uint32_t[dimension_number * count];
254  uint32_t *uint_ptr1;
255  T min, range;
256 
257  for (i = 0; i < dimension_number; i++) {
258  uint_ptr1 = uint_point + i * count;
259  ptr1 = point + i * count;
260  min = minmax[i];
261  range = minmax[i + dimension_number] - min;
262  for (j = 0; j < count; j++)
263  uint_ptr1[j] = (ptr1[j] - min) / range * UINT32_MAX;
264  }
265 
266  delete[] minmax;
267 
268  // Now uint_ptr1 contains initial coordinates, rescaled to range
269  // [0, UINT32_MAX] and converted to uint32_t type to use special radix sort
270  // Prepare indexes to store sort order
271  auto order = new int[count];
272  for (j = 0; j < count; j++)
273  order[j] = j;
274  SortArray(uint_point, count);
275 
276  auto new_point = new T[dimension_number * count];
277  auto new_z = new T[count];
278 
279  for (j = 0; j < count; j++) {
280  for (i = 0; i < dimension_number; i++)
281  new_point[count * i + j] = point[count * i + order[j]];
282  new_z[j] = apZ[order[j]];
283  }
284 
285  for (i = 0; i < count; i++) {
286  aLocations.GetLocationX()[i] = new_point[i];
287  aLocations.GetLocationY()[i] = new_point[i + count];
288  apZ[i] = new_z[i];
289  }
290 
291  delete[] new_point;
292  delete[] point;
293  delete[] new_z;
294  delete[] uint_point;
295  delete[] order;
296 
297  return 0;
298 }
