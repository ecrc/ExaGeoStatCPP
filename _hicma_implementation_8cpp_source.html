33 void HicmaImplementation<T>::SetModelingDescriptors(ExaGeoStatData<T> &aData, Configurations &aConfigurations) {
34 
35  int N = aConfigurations.GetProblemSize();
36  int lts = aConfigurations.GetLowTileSize();
37  int p_grid = aConfigurations.GetPGrid();
38  int q_grid = aConfigurations.GetQGrid();
39  bool is_OOC = aConfigurations.GetIsOOC();
40  int max_rank = aConfigurations.GetMaxRank();
41 
42  // Set the floating point precision based on the template type
43  FloatPoint float_point;
44  if (sizeof(T) == SIZE_OF_FLOAT) {
45  float_point = EXAGEOSTAT_REAL_FLOAT;
46  } else if (sizeof(T) == SIZE_OF_DOUBLE) {
47  float_point = EXAGEOSTAT_REAL_DOUBLE;
48  } else {
49  throw runtime_error("Unsupported for now!");
50  }
51 
52  int MBD = lts;
53  int NBD = lts;
54  int MD = N;
55  int ND = MBD;
56  aData.GetDescriptorData()->SetDescriptor(common::HICMA_DESCRIPTOR, DESCRIPTOR_CD, is_OOC, nullptr, float_point, MBD,
57  NBD, MBD * NBD, MD, ND, 0, 0, MD, ND, p_grid, q_grid);
58  int MBUV = lts;
59  int NBUV = 2 * max_rank;
60  int MUV;
61  int N_over_lts_times_lts = N / lts * lts;
62  if (N_over_lts_times_lts < N) {
63  MUV = N_over_lts_times_lts + lts;
64  } else if (N_over_lts_times_lts == N) {
65  MUV = N_over_lts_times_lts;
66  } else {
67  throw runtime_error("This case can't happens, N need to be >= lts*lts");
68  }
69  int expr = MUV / lts;
70  int NUV = 2 * expr * max_rank;
71  aData.GetDescriptorData()->SetDescriptor(common::HICMA_DESCRIPTOR, DESCRIPTOR_CUV, is_OOC, nullptr, float_point,
72  MBUV, NBUV, MBUV * NBUV, MUV, NUV, 0, 0, MUV, NUV, p_grid, q_grid);
73  auto *HICMA_descCUV = aData.GetDescriptorData()->GetDescriptor(DescriptorType::HICMA_DESCRIPTOR,
74  DescriptorName::DESCRIPTOR_CUV).hicma_desc;
75  int MBrk = 1;
76  int NBrk = 1;
77  int Mrk = HICMA_descCUV->mt;
78  int Nrk = HICMA_descCUV->mt;
79  aData.GetDescriptorData()->SetDescriptor(common::HICMA_DESCRIPTOR, DESCRIPTOR_CRK, is_OOC, nullptr, float_point,
80  MBrk, NBrk, MBrk * NBrk, Mrk, Nrk, 0, 0, Mrk, Nrk, p_grid, q_grid);
81  aData.GetDescriptorData()->SetDescriptor(common::HICMA_DESCRIPTOR, DESCRIPTOR_Z_COPY, is_OOC, nullptr, float_point,
82  lts, lts, lts * lts, N, 1, 0, 0, N, 1, p_grid, q_grid);
83 }

225  dot_product = *product;

228 
229  LOGGER(iter_count + 1 << " - Model Parameters (", true)

231  fprintf(aConfigurations.GetFileLogPath(), " %3d- Model Parameters (", iter_count + 1);
232  }

234  (aConfigurations.GetKernelName() == "bivariate_matern_parsimonious2_profile")) {

265 
266  aData.SetMleIterations(aData.GetMleIterations() + 1);

352 
353  // Check for initialize the Hicma context.
354  if (!this->mpContext) {

359  this->ExaGeoStatOptionsInit(&options, this->mpContext, apSequence, apRequest);

361  int m;

365  struct starpu_codelet *cl = &this->cl_dmdet;
366 
367  for (m = 0; m < Z->mt; m++) {
368  temp = m == Z->mt - 1 ? Z->m - m * Z->mb : Z->mb;

372  STARPU_R, ExaGeoStatDataGetAddr(Z, m, 0),
373  STARPU_RW, ExaGeoStatDataGetAddr(det, 0, 0),

375  }
376  this->ExaGeoStatOptionsFree(&options);
377  this->ExaGeoStatOptionsFinalize(&options, (HICMA_context_t *)
378  this->mpContext);
