292  const static std::set<std::string> availableKernels = []() {
293  // This set stores the kernel names.
294  std::set<std::string> kernelNames;
295  // This string stores the directory path where the kernel files are located.
296  const std::string directoryPath = KERNELS_PATH;
297  // This loop iterates through all the files in the directory and extracts the kernel names.
298  for (const auto &entry: std::filesystem::directory_iterator(directoryPath)) {
299  // This checks if the current entry is a regular file.
300  if (entry.is_regular_file()) {
301  // This string stores the filename of the current entry.
302  const std::string filename = entry.path().filename().string();
303  // This string stores the file extension of the current entry.
304  const std::string extension = std::filesystem::path(filename).extension().string();
305  // This string stores the kernel name extracted from the filename.
306  const std::string kernelName = filename.substr(0, filename.size() - extension.size());
307  // This adds the kernel name to the kernelNames set.
308  kernelNames.insert(kernelName);
309  // This blo/ck of code converts the kernel name to lowercase and adds underscores before each capital letter.
310  std::string lowercaseName;
311  for (std::size_t i = 0; i < kernelName.size(); ++i) {
312  if (std::isupper(kernelName[i])) {
313  // Avoid adding _ in the beginning of the name.
314  if (i != 0) {
315  lowercaseName += '_';
316  }
317  lowercaseName += static_cast<char>(std::tolower(kernelName[i]));
318  } else {
319  lowercaseName += kernelName[i];
320  }
321  }
322  // This adds the lowercase kernel name to the kernelNames set.
323  kernelNames.insert(lowercaseName);
324  }
325  }
326  return kernelNames;
327  }();
