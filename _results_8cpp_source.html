44 void Results::PrintEndSummary() {
45 
46  Verbose temp = exageostat::configurations::Configurations::GetVerbosity();
47  exageostat::configurations::Configurations::SetVerbosity(STANDARD_MODE);
48  LOGGER("")
49  LOGGER("********************SUMMARY**********************")
50 
51  auto locations_number = mGeneratedLocationsNumber;
52  if (locations_number > 0) {
53  LOGGER("---- Data Generation Results ----")
54  if (mIsSynthetic) {
55  LOGGER(" #Synthetic Dataset")
56  } else {
57  LOGGER(" #Real Dataset")
58  }
59  LOGGER(" #Number of Locations: " << locations_number)
60  if (mIsLogger && mIsSynthetic) {
61  LOGGER(" #Data is written to file (", true)
62  if (mLoggerPath.empty()) {
63  mLoggerPath = LOG_PATH;
64  }
65  LOGGER_PRECISION(mLoggerPath << ").")
66  LOGGER("")
67  }
68  LOGGER(" #Total Data Generation Execution Time: " << mExecutionTimeDataGeneration)
69  LOGGER(" #Total Data Generation Gflop/s: " << mFlopsDataGeneration)
70  LOGGER("")
71  }
72  if (mMLEIterations > 0) {
73  LOGGER("---- Data Modeling Results ----")
74  LOGGER(" #Number of MLE Iterations till reach Maximum: " << mMLEIterations)
75  LOGGER(" #Found Maximum Theta at: ", true)
76  for (double i: mMaximumTheta) {
77  LOGGER_PRECISION(i << " ", 8)
78  }
79  LOGGER("")
80  LOGGER(" #Final Log Likelihood value: " << mLogLikValue)
81  LOGGER(" #Average Time Modeling per Iteration: " << this->GetAverageModelingExecutionTime())
82  LOGGER(" #Average Flops per Iteration: " << this->GetAverageModelingFlops())
83  LOGGER(" #Total MLE Execution time: " << mTotalModelingExecutionTime)
84  LOGGER(" #Total MLE Gflop/s: " << mTotalModelingFlops)
85  LOGGER("")
86  }
87 
88  if (mZMiss > 0) {
89  LOGGER("---- Data Prediction Results ----")
90  LOGGER(" #Number of Missing Observations: " << mZMiss)
91  if (mMSPEError > 0) {
92  LOGGER(" #MSPE")
93  LOGGER(" #MSPE Prediction Execution Time: " << mExecutionTimeMSPE)
94  LOGGER(" #MSPE Gflop/s: " << mFlopsMSPE)
95  LOGGER(" #Mean Square Error MSPE: " << mMSPEError)
96 
97  }
98  if (!mIDWError.empty()) {
99  LOGGER(" #IDW")
100  LOGGER(" #IDW Error: ( ", true)
101  for (int i = 0; i < 3; i++) {
102  LOGGER_PRECISION(mIDWError[i] << " ", 8)
103  }
104  LOGGER_PRECISION(").")
105  LOGGER("")
106  }
107  if (mMLOE > 0 || mMMOM > 0) {
108  LOGGER(" #MLOE MMOM")
109  LOGGER(" #MLOE: " << mMLOE)
110  LOGGER(" #MMOM: " << mMMOM)
111  LOGGER(" #MLOE-MMOM Execution Time: " << mExecutionTimeMLOEMMOM)
112  LOGGER(" #MLOE-MMOM Matrix Generation Time: " << mGenerationTimeMLOEMMOM)
113  LOGGER(" #MLOE-MMOM Cholesky Factorization Time: " << mFactoTimeMLOEMMOM)
114  LOGGER(" #MLOE-MMOM Loop Time: " << mLoopTimeMLOEMMOM)
115  LOGGER(" #MLOE-MMOM Number of flops: " << mFlopsMLOEMMOM)
116  LOGGER("")
117  }
118  }
119  if(mFisher00 != 0){
120  LOGGER(" #Fisher")
121  LOGGER(" #Sd For Sigma2: " << mFisher00)
122  LOGGER(" #Sd For Alpha: " << mFisher11)
123  LOGGER(" #Sd For Nu: " << mFisher22)
124  LOGGER(" #Fisher Execution Time: " << mTotalFisherTime)
125  LOGGER("")
126  }
127  LOGGER("*************************************************")
128  exageostat::configurations::Configurations::SetVerbosity(temp);
129 }
